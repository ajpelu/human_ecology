lmer.glmulti <- function(formula, data, random = "", always = "", ...) {
    newf <- formula
    newf[[3]] <- substitute(f + a + r,
                            list(f = newf[[3]], 
                                 r = reformulate(random)[[2]],
                                 a = reformulate(always)[[2]]))
    lmer(newf, data = data,
         REML = FALSE, ...)
}
